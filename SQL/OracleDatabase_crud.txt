-- Comandos para Oracle Database

-- 1. Create Database (Criar Banco de Dados)
-- Em Oracle, a criação de um banco de dados é mais complexa e geralmente feita pelo DBA.
-- Aqui, vamos assumir que estamos conectados a um banco de dados existente.

-- 2. Create Table (Criar Tabela)
CREATE TABLE usuarios (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome VARCHAR2(100),
    email VARCHAR2(100),
    idade NUMBER
);

-- 3. Create (Inserir Dados)
INSERT INTO usuarios (nome, email, idade) VALUES ('João Silva', 'joao@example.com', 30);
INSERT INTO usuarios (nome, email, idade) VALUES 
('Maria Oliveira', 'maria@example.com', 25),
('Carlos Souza', 'carlos@example.com', 28);

-- 4. Read (Consultar Dados)
SELECT * FROM usuarios;
SELECT * FROM usuarios WHERE idade > 25;
SELECT * FROM usuarios WHERE email = 'joao@example.com';

-- 5. Update (Atualizar Dados)
UPDATE usuarios SET idade = 31 WHERE email = 'joao@example.com';
UPDATE usuarios SET idade = 26 WHERE email = 'maria@example.com';
UPDATE usuarios SET idade = 29 WHERE email = 'carlos@example.com';

-- 6. Delete (Remover Dados)
DELETE FROM usuarios WHERE email = 'joao@example.com';
DELETE FROM usuarios WHERE idade < 30;
